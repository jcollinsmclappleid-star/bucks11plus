2) TYPES + VALIDATION (STRICT)
===========================================================

Create /lib/content/types.ts and export strict TS types for render_config schemas.

-----------------------------------------------------------
2.1 Common types
-----------------------------------------------------------

export type RenderType = "text" | "svg" | "chart";

export type SvgStroke = {
  strokeWidth: number;         // e.g. 3
  stroke: string;              // use neutral token e.g. "#111827"
  fill: "none" | "solid";
  dashed?: boolean;
  opacity?: number;
};

export type SvgShape = "circle" | "square" | "triangle" | "pentagon" | "arrow" | "star";

export type SvgElement =
  | { type: "shape"; shape: SvgShape; x: number; y: number; size: number; rotation: number; style: SvgStroke }
  | { type: "line"; x1: number; y1: number; x2: number; y2: number; style: SvgStroke }
  | { type: "dot"; x: number; y: number; r: number; style: SvgStroke };

export type SvgFrame = {
  elements: SvgElement[];
};

-----------------------------------------------------------
2.2 NVR configs
-----------------------------------------------------------

export type NvrSequenceConfig = {
  kind: "nvr.sequence";
  frames: SvgFrame[];              // includes final unknown as empty frame or placeholder
  questionIndex: number;           // which frame is "?"
  answerOptions: SvgFrame[];       // 4 options
};

export type NvrTransformConfig = {
  kind: "nvr.transform";
  promptFrames: SvgFrame[];        // e.g. A -> B, then C -> ?
  answerOptions: SvgFrame[];
};

export type NvrClassificationConfig = {
  kind: "nvr.classification";
  group: SvgFrame[];               // 4-6 frames, one is odd
  answerOptions: SvgFrame[];       // can be labels or frames; use frames for consistency
};

-----------------------------------------------------------
2.3 Chart configs (Skill 18)
-----------------------------------------------------------

export type ChartBarConfig = {
  kind: "chart.bar";
  title: string;
  xLabels: string[];
  values: number[];
  yMax: number;
  questionText: string;            // e.g. "How many more ...?"
};

export type ChartLineConfig = {
  kind: "chart.line";
  title: string;
  xLabels: string[];
  values: number[];
  yMax: number;
  questionText: string;
};

export type ChartTableConfig = {
  kind: "chart.table";
  title: string;
  headers: string[];
  rows: (string | number)[][];
  questionText: string;
};

export type RenderConfig =
  | NvrSequenceConfig
  | NvrTransformConfig
  | NvrClassificationConfig
  | ChartBarConfig
  | ChartLineConfig
  | ChartTableConfig;
